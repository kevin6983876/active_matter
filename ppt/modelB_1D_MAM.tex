\section{Hamiltonian}
\begin{frame}{Hamiltonian: Model A}
    (Original \texttt{modified\_Ginzburg\_Landau\_MAM.py})
    \begin{equation}
        H = \sum_{\mathrm{boxes}} \left[ \big( D\nabla^2\rho + \rho - \rho^3 + \kappa\langle\rho^2\rangle \big) \theta \;+\; \frac{a}{2}\,\theta^2 \right].
    \end{equation}
    \begin{itemize}
        \item Laplacian: discrete 3-point stencil via \texttt{np.roll}.
        \item Nonlocal coupling $\kappa\langle\rho^2\rangle$; noise term $a\theta^2/2$ is \textbf{not} conserved.
    \end{itemize}
\end{frame}

\begin{frame}{Hamiltonian: Model B}
    (\texttt{MAM\_modelB.py})
    \begin{equation}
        H = \sum_{\mathrm{boxes}} \left[ \nabla^2\big( -D\nabla^2\rho - \rho + \rho^3 \big)\,\theta \;-\; \frac{a}{2}\,\theta\,\nabla^2\theta \right].
    \end{equation}
    \begin{itemize}
        \item Deterministic part: $\nabla^2\mu$ with $\mu = -D\nabla^2\rho - \rho + \rho^3$ (chemical potential).
        \item Noise part: $-\frac{a}{2}\theta\nabla^2\theta$ is \textbf{conserved} (gradient form).
        \item No $\kappa$; no nonlocal coupling.
        \item Laplacian($\nabla^2$) implemented in \textbf{Fourier space}: \texttt{lap(field)} = IFFT($-k^2\cdot$FFT(field)); uses \texttt{k2}, and \texttt{k4} for biharmonic terms ($\nabla^4$) where needed.
    \end{itemize}
\end{frame}

\section{Laplacian and derivatives}
\begin{frame}{Laplacian: Model A vs Model B}
    \textbf{Model A:} real-space finite difference
    \begin{equation}
        \nabla^2 f \;\to\; \frac{f_{i+1} + f_{i-1} - 2f_i}{h^2}, \quad \texttt{np.roll}.
    \end{equation}
    \vspace{0.5em}
    \textbf{Model B:} Fourier space
    \begin{equation}
        \nabla^2 f \;\to\; \mathcal{F}^{-1}(-k^2\,\mathcal{F}(f)), \quad \texttt{k\_vec}, \; \texttt{k2} = k^2, \; \texttt{k4} = k^4.
    \end{equation}
    \texttt{lap(field)} FFTs along the spatial axis (last axis), multiplies by $-k^2$, then IFFTs. Used for both $\rho$ and $\theta$ in $H$ and in $\partial H/\partial\rho$, $\partial H/\partial\theta$.
\end{frame}

\begin{frame}{$\partial H/\partial\rho$ and $\partial H/\partial\theta$: Model B}
    In Model B the derivatives are (in the code):
    \begin{align*}
        \frac{\partial H}{\partial \rho} &= -D\,\nabla^2(\nabla^2\theta) - (1-3\rho^2)\,\nabla^2\theta, \\
        \frac{\partial H}{\partial \theta} &= \nabla^2(-D\nabla^2\rho - \rho + \rho^3) - a\,\nabla^2\theta.
    \end{align*}
    Implemented using \texttt{lap(rho)}, \texttt{lap(theta)}, and \texttt{lap(lap\_rho)} etc.; no \texttt{np.roll}. Reaction terms \texttt{reaction\_U} and \texttt{reaction\_V} are then built as in Model A (same $U,V$ split), then FFT in the copy direction and tridiagonal solve.
\end{frame}

\section{Tridiagonal solve and IMEX}
\begin{frame}{Tridiagonal solve: Model A vs Model B (matrix form)}
    In Fourier space, for each wavenumber $k$ we solve a linear system in the \textbf{path (copy)} index $j$. Let $r = \Delta\tau/\Delta s$ and $\eta_k = \gamma\,\Delta\tau\,D\,k^4$ (IMEX scalar).
    \vspace{0.4em}
    \textbf{Example:} $V$-field (lower triangular $B_k$). \textbf{Model A:}
    \[
    B_k\,\mathbf{V}_k = \mathbf{b}_k,
    \quad
    \mathbf{b}_k = \begin{bmatrix} \mathrm{BC}_0 \\ V_1^{\mathrm{old}} + \Delta\tau\,F_1 \\ \vdots \\ V_{N-1}^{\mathrm{old}} + \Delta\tau\,F_{N-1} \end{bmatrix}.
    \]
    $B_k$ is lower triangular (e.g.\ diagonal $1+r$, subdiagonal $-r$; first row encodes boundary). 
    \vspace{0.3em}
    \textbf{Model B (IMEX):} Add $\eta_k I$ to the matrix and $\eta_k\,\mathbf{V}_k^{\mathrm{old}}$ to the RHS:
    \[
    (B_k + \eta_k I)\,\mathbf{V}_k = \mathbf{b}_k + \eta_k\,\mathbf{V}_k^{\mathrm{old}}.
    \]
    Diagonal of the solved matrix becomes $(1+r+\eta_k)$ etc. Same \texttt{solve\_triangular(B\_..., ..., lower=True)}; only entries and RHS differ.
\end{frame}

\begin{frame}{IMEX: what it does \& why physics is unchanged}
    In Fourier space the biharmonic term is $\sim D k^4$. Pure explicit stepping gives $U_{\mathrm{new}} \approx (1 - \Delta\tau D k^4) U_{\mathrm{old}}$ $\Rightarrow$ high-$k$ blows up. IMEX puts $\eta_k U_{\mathrm{new}}$ on the \textbf{left} (diagonal) and $\eta_k U_{\mathrm{old}}$ on the \textbf{right}. Then $U_{\mathrm{new}} = \mathrm{RHS}/(1+\eta_k)$: high-$k$ is \textbf{damped}. $\gamma$ (e.g.\ 2.0) controls the strength.
    \vspace{0.5em}

    At \textbf{convergence}, $\mathbf{V}_k = \mathbf{V}_k^{\mathrm{old}}$, so the implicit term $\eta_k\mathbf{V}_k$ and the explicit term $\eta_k\mathbf{V}_k^{\mathrm{old}}$ in $(B_k+\eta_k I)\mathbf{V}_k = \mathbf{b}_k + \eta_k\mathbf{V}_k^{\mathrm{old}}$ cancel. The limit satisfies the true equation; the stabilizer only affects the \emph{path} to the minimum, not the minimum itself.
\end{frame}

\section{Boundary conditions and mass}
\begin{frame}{Endpoints $\rho_1$, $\rho_2$: Model A vs Model B}
    \textbf{Model A:} Homogeneous endpoints. $\rho_1 = \rho_- \cdot \mathbf{1}$, $\rho_2 = \rho_+ \cdot \mathbf{1}$ (all boxes equal); $\rho_\pm$ from roots of cubic (with $\kappa$). \texttt{solRho = np.sort(np.roots([-1, kappa, 1, 0]))}.
    \vspace{0.5em}
    
    \textbf{Model B:} \textbf{Inhomogeneous} endpoints (spatial profile). E.g.\ \texttt{rho1[0]=solRho[0]}, \texttt{rho1[L//2]=solRho[2]} (``left low, middle high'') and \texttt{rho2} the other way; \texttt{solRho = [-1, 0, 1]} fixed. Path connects two different spatial patterns with the same total mass.
\end{frame}

\begin{frame}{Mass conservation in Model B}
    Model B conserves total mass $\int \rho$. In the code, after each half-step (after updating $U$ or $V$ and recovering $\rho$, $\theta$):
    \begin{itemize}
        \item \texttt{target\_mass = np.mean(rho1)} (or rho1.real for consistency).
        \item \texttt{rho = rho - np.mean(rho, axis=1, keepdims=True) + target\_mass}.
        \item \texttt{theta = theta - np.mean(theta, axis=1, keepdims=True)} (zero-mean $\theta$).
    \end{itemize}
    Then re-enforce boundary: \texttt{rho[0,:]=rho1}, \texttt{rho[Ncopy-1,:]=rho2}. Initial path bump is also mean-subtracted: \texttt{bump = bump - np.mean(bump)}.
\end{frame}

\section{Other differences}
\begin{frame}{Parameters and extra output (Model B)}
    \begin{itemize}
        \item \textbf{Parameters:} Model B uses \texttt{L=2}, \texttt{Ncopy=400}, \texttt{Tmax=2}, \texttt{iterations=2000}, \texttt{D=0.001}, no \texttt{kappa}; \texttt{solRho = [-1, 0, 1]}.
        \item \textbf{Lagrangian\_Theo:} Model B defines \texttt{Lagrangian\_Theo(h, rho, theta)} = $-0.5\,a\,\theta\,\nabla^2\theta$ (theoretical Lagrangian density for conserved noise). Plot includes $L_{\mathrm{th}}$ alongside $L$ and $H$.
    \end{itemize}
\end{frame}

\section{Summary}
\begin{frame}{Summary: Model B vs Model A}
    \begin{itemize}
        \item \textbf{Physics:} Model B = conserved dynamics (gradient form + conserved noise); Model A = non-conserved with $\kappa$ and $\theta^2$ noise.
        \item \textbf{Hamiltonian:} Model B uses $\nabla^2\mu\,\theta$ and $-\frac{a}{2}\theta\nabla^2\theta$; Model A uses $(D\nabla^2\rho + \rho - \rho^3 + \kappa\langle\rho^2\rangle)\theta + \frac{a}{2}\theta^2$.
        \item \textbf{Laplacian:} Model A = real-space stencil; Model B = FFT (\texttt{lap}, \texttt{k2}, \texttt{k4}).
        \item \textbf{Tridiagonal:} Model B adds IMEX stabilization (\texttt{gamma}, \texttt{k4}) to diagonal and RHS.
        \item \textbf{Endpoints \& mass:} Model B uses inhomogeneous $\rho_1,\rho_2$ and enforces mass conservation each half-step; Model A uses homogeneous endpoints, no mass correction.
    \end{itemize}
\end{frame}
